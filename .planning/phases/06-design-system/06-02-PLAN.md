NOTE: Import noto-sans-devanagari with `/wght.css` suffix to get weight axis only with proper unicode-range subsetting (per Research Pitfall 4).

**HTML structure:**
```html
<!doctype html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <meta name="description" content={description} />

    <!-- FOUC prevention: apply theme class before paint (per D-01, Research Pattern 4) -->
    <script is:inline>
      (function() {
        var VALID = ['dark', 'light', 'sepia'];
        var saved = localStorage.getItem('theme');
        var theme = (saved && VALID.indexOf(saved) !== -1) ? saved : 'dark';
        document.documentElement.classList.add(theme);
      })();
    </script>

    <title>{title} | The Bhagavad Gita</title>
  </head>
  <body class="min-h-screen flex flex-col bg-[--color-bg] text-[--color-text] transition-colors duration-300">

    <!-- Theme toggle button (per D-05 — lives in header area) -->
    <div class="fixed top-4 right-4 z-50">
      <button
        id="theme-toggle"
        aria-label="Toggle display theme"
        title="Toggle display theme"
        class="p-2 rounded-lg bg-[--color-bg-raised] border border-[--color-border] text-[--color-text-2] hover:text-[--color-accent] transition-colors text-sm font-[--font-ui]"
      >
      </button>
    </div>

    <slot />

    <footer class="border-t border-[--color-border] py-10 text-center mt-auto">
      <p class="font-[--font-devanagari] text-sm text-[--color-text-3] mb-2" style="font-size: 110%;">
        कर्मण्येवाधिकारस्ते मा फलेषु कदाचन
      </p>
      <p class="text-xs text-[--color-text-4] tracking-wide">
        Multi-tradition synthesis &middot; IAST transliteration &middot; Open source
      </p>
      <a href={`${import.meta.env.BASE_URL}learn-sanskrit/`} class="inline-block mt-3 text-xs text-[--color-accent] hover:text-[--color-text-2] transition-colors tracking-wide">
        Learn Sanskrit in 14 Days &rarr;
      </a>
    </footer>

    <!-- Theme toggle logic + transition re-apply (vanilla JS per Research Pattern 6) -->
    <script>
      const THEMES: readonly string[] = ['dark', 'light', 'sepia'] as const;
      const ICONS: Record<string, string> = { dark: '\u263D', light: '\u2600', sepia: '\u{1F4DC}' };
      const LABELS: Record<string, string> = { dark: 'Dark', light: 'Light', sepia: 'Sepia' };

      function getTheme(): string {
        const saved = localStorage.getItem('theme');
        return (saved && THEMES.includes(saved)) ? saved : 'dark';
      }

      function applyTheme(): void {
        const theme = getTheme();
        document.documentElement.classList.remove(...THEMES);
        document.documentElement.classList.add(theme);
        updateToggleButton(theme);
      }

      function cycleTheme(): void {
        const current = getTheme();
        const idx = THEMES.indexOf(current);
        const next = THEMES[(idx + 1) % THEMES.length];
        localStorage.setItem('theme', next);
        document.documentElement.classList.remove(...THEMES);
        document.documentElement.classList.add(next);
        updateToggleButton(next);
      }

      function updateToggleButton(theme: string): void {
        const btn = document.getElementById('theme-toggle');
        if (btn) {
          btn.textContent = `${ICONS[theme] || ''} ${LABELS[theme] || ''}`;
          btn.setAttribute('aria-label', `Current theme: ${LABELS[theme]}. Click to switch.`);
        }
      }

      // Wire up toggle button click
      function init(): void {
        const btn = document.getElementById('theme-toggle');
        btn?.addEventListener('click', cycleTheme);
        applyTheme();
      }

      // Run on initial load and after each Astro page transition
      document.addEventListener('astro:page-load', init);
      // Re-apply theme immediately after swap (before paint) to prevent FOUC
      document.addEventListener('astro:after-swap', applyTheme);
    </script>
  </body>
</html>
```

**Key details:**
- The `is:inline` script in `<head>` validates localStorage value against a whitelist (`['dark', 'light', 'sepia']`). Unknown values default to 'dark'. This prevents XSS-injected localStorage values from becoming class names.
- The `<body>` uses semantic tokens: `bg-[--color-bg]`, `text-[--color-text]`.
- Footer hardcoded colors replaced with semantic tokens: `--color-border`, `--color-text-3`, `--color-text-4`, `--color-accent`.
- Devanagari text in footer uses `font-[--font-devanagari]` with `style="font-size: 110%;"` (per D-08 — 110% scale).
- Theme toggle is positioned fixed top-right with z-50 — Phase 7 will move it into the proper header component. This is a temporary placement.
- The `astro:after-swap` listener re-applies theme class immediately after ClientRouter replaces documentElement (per Research Pattern 4, Pitfall 1).
- The `astro:page-load` listener re-initializes the toggle button event listener after navigation (per Research Pitfall 3).
- meta theme-color removed (was hardcoded to #FAF7F2 — theme-dependent, will be handled in Phase 10 SEO work).
